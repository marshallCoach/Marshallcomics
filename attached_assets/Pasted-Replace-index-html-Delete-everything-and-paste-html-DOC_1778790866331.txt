// ── TABS ──
document.querySelectorAll('.tab-btn').forEach(btn => {
  btn.addEventListener('click', () => {
    document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
    document.querySelectorAll('.tab-content').forEach(t => t.classList.remove('active'));
    btn.classList.add('active');

// ── SHARED: toggle expand ──
function toggle(el, cls) {
  el.classList.toggle(cls);
  const exp = el.querySelector('.card-expand, .lcard-expand, .show-expand');
  if (exp) exp.classList.toggle('open');
}

// ── SHARED: platform badge ──
function platBadge(p) {
  const map = { 'WHATNOT':'bwn', 'EBAY':'beb', 'HERITAGE':'bhe', 'WHATNOT+HERITAGE':'bhe' };
  const cls = map[(p||'').toUpperCase()] || '';
  return cls ? `<span class="badge ${cls}">${p}</span>` : '';
}

// ── SHARED: no results ──
function noRes(grid) { grid.innerHTML = '<div class="no-res">NO BOOKS MATCH</div>'; }

// ──────────────────────────────────────────────
// ORIGINAL COLLECTION
// ──────────────────────────────────────────────
function renderOrig(data) {
  const g = document.getElementById('o-grid');
  document.getElementById('o-count').textContent = `${data.length} book${data.length!==1?'s':''}`;
  if (!data.length) { noRes(g); return; }
  g.innerHTML = data.map(c => {
    const isKey = (c.Key||'').toUpperCase()==='YES';
    const isSigned = (c.Signed||'').toUpperCase()==='YES';
    const isCGC = (c.CGC_Worth||'').toUpperCase()==='YES';
    const tf = (c.Terrificon||'').trim();
    return `<div class="comic-card" onclick="toggle(this,'open')">
      <div class="card-title">${c.Title||'Untitled'}</div>
      <div class="card-sub">${c.Publisher||''} #${c.Issue||''} · ${c.Year||''} · ${c.Era||''}</div>
      <div class="badges">
        ${isKey?'<span class="badge bk">KEY</span>':''}
        ${isSigned?`<span class="badge bs">✍ ${(c.Signed_By||'').substring(0,20)}</span>`:''}
        ${c.Era?`<span class="badge be">${c.Era}</span>`:''}
        ${platBadge(c.Platform)}
        ${isCGC?'<span class="badge bc">CGC ✓</span>':''}
        ${tf?'<span class="badge bt">Terrificon</span>':''}
      </div>
      ${c.Value_NM?`<div class="card-value">NM: <span class="v">${c.Value_NM}</span>${c.Value_VF?' · VF: '+c.Value_VF:''}</div>`:''}
      ${c.Whatnot_Pitch?`<div class="card-pitch">${c.Whatnot_Pitch.substring(0,170)}${c.Whatnot_Pitch.length>170?'…':''}</div>`:''}
